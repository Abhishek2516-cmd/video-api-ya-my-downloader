async function startProcess() { let videoUrl = document.getElementById('link').value; let status = document.getElementById('status'); if(!videoUrl) { alert("Pehle link toh dalo!"); return; } // Interstitial Ad Trigger (Sirf App ke liye) if(window.Android) { try { window.Android.showInterstitial(); } catch(e) { console.log("Ad Error"); } } status.innerHTML = "🔍 Checking Link... Please wait."; status.style.color = "yellow"; try { // API URL ko sahi format mein call karna const apiUrl = `https://video-api-ya-my-downloader-2.onrender.com/dl?url=${encodeURIComponent(videoUrl)}`; const response = await fetch(apiUrl); const data = await response.json(); if(data.url) { status.innerHTML = "✅ Video Found! Starting Download..."; status.style.color = "#00e676"; window.location.href = data.url; } else { status.innerHTML = "❌ Error: Video link not supported or expired."; status.style.color = "red"; } } catch (err) { console.error(err); status.innerHTML = "⚠️ API Error: Make sure your Render server is awake."; status.style.color = "orange"; } }
